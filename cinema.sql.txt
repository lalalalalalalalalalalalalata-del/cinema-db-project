CREATE TABLE genres (
    id SERIAL PRIMARY KEY,
    name VARCHAR(50) UNIQUE NOT NULL
);

-- Фильмы
CREATE TABLE films (
    id SERIAL PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    description TEXT,
    genre_id INTEGER REFERENCES genres(id) ON DELETE SET NULL,
    duration_minutes INTEGER NOT NULL CHECK (duration_minutes > 0),
    age_rating INTEGER CHECK (age_rating BETWEEN 0 AND 21),
    poster_url VARCHAR(500),
    release_date DATE,
    country VARCHAR(100),
    created_at TIMESTAMP DEFAULT NOW()
);

-- Залы
CREATE TABLE halls (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    rows_count INTEGER NOT NULL CHECK (rows_count > 0),
    seats_per_row INTEGER NOT NULL CHECK (seats_per_row > 0),
    hall_type VARCHAR(50) CHECK (hall_type IN ('Standard', 'VIP', 'IMAX')) DEFAULT 'Standard',
    description TEXT
);

-- Места в залах
CREATE TABLE seats (
    id SERIAL PRIMARY KEY,
    hall_id INTEGER NOT NULL REFERENCES halls(id) ON DELETE CASCADE,
    row_number INTEGER NOT NULL,
    seat_number INTEGER NOT NULL,
    seat_type VARCHAR(50) CHECK (seat_type IN ('Standard', 'VIP', 'Accessible')) DEFAULT 'Standard',
    UNIQUE (hall_id, row_number, seat_number)
);

-- Пользователи
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    full_name VARCHAR(200) NOT NULL,
    email VARCHAR(200) UNIQUE NOT NULL,
    phone VARCHAR(20),
    birth_date DATE,
    password_hash VARCHAR(200) NOT NULL,
    registered_at TIMESTAMP DEFAULT NOW(),
    is_active BOOLEAN DEFAULT TRUE
);

-- Сеансы
CREATE TABLE sessions (
    id SERIAL PRIMARY KEY,
    film_id INTEGER NOT NULL REFERENCES films(id) ON DELETE CASCADE,
    hall_id INTEGER NOT NULL REFERENCES halls(id) ON DELETE CASCADE,
    start_time TIMESTAMP NOT NULL,
    end_time TIMESTAMP NOT NULL,
    base_price DECIMAL(10,2) NOT NULL CHECK (base_price >= 0),
    created_at TIMESTAMP DEFAULT NOW(),
    CHECK (start_time < end_time)
);

-- Билеты
CREATE TABLE tickets (
    id SERIAL PRIMARY KEY,
    session_id INTEGER NOT NULL REFERENCES sessions(id) ON DELETE CASCADE,
    seat_id INTEGER NOT NULL REFERENCES seats(id) ON DELETE CASCADE,
    user_id INTEGER REFERENCES users(id) ON DELETE SET NULL,
    price DECIMAL(10,2) NOT NULL,
    status VARCHAR(50) NOT NULL DEFAULT 'booked' 
        CHECK (status IN ('booked', 'paid', 'cancelled', 'refunded')),
    sale_time TIMESTAMP DEFAULT NOW(),
    payment_method VARCHAR(50),
    UNIQUE (session_id, seat_id)
);

-- Лог изменений статусов билетов
CREATE TABLE ticket_status_log (
    id SERIAL PRIMARY KEY,
    ticket_id INTEGER NOT NULL REFERENCES tickets(id) ON DELETE CASCADE,
    old_status VARCHAR(50),
    new_status VARCHAR(50) NOT NULL,
    changed_at TIMESTAMP DEFAULT NOW(),
    changed_by VARCHAR(100) DEFAULT 'system'
);


2. функции и триггеры
Триггер 1: Автовычисление времени окончания сеанса
sql
CREATE OR REPLACE FUNCTION set_session_end_time()
RETURNS TRIGGER AS $$
DECLARE
    film_duration INTEGER;
BEGIN
    -- Получаем длительность фильма
    SELECT duration_minutes INTO film_duration
    FROM films WHERE id = NEW.film_id;
    
    IF film_duration IS NULL THEN
        RAISE EXCEPTION 'Film not found or duration is missing';
    END IF;
    
    -- Вычисляем end_time = start_time + duration_minutes
    NEW.end_time := NEW.start_time + (film_duration || ' minutes')::INTERVAL;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_set_session_end_time
    BEFORE INSERT OR UPDATE OF start_time, film_id ON sessions
    FOR EACH ROW
    EXECUTE FUNCTION set_session_end_time();


Триггер 2: Проверка доступности места при продаже/бронировании
sql
CREATE OR REPLACE FUNCTION check_seat_availability()
RETURNS TRIGGER AS $$
BEGIN
    -- Проверяем, есть ли уже билет на это место в том же сеансе со статусом, отличным от 'cancelled'
    IF EXISTS (
        SELECT 1 FROM tickets
        WHERE session_id = NEW.session_id
          AND seat_id = NEW.seat_id
          AND status IN ('booked', 'paid')
    ) THEN
        RAISE EXCEPTION 'Место % уже занято в данном сеансе', NEW.seat_id;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_check_seat_availability
    BEFORE INSERT ON tickets
    FOR EACH ROW
    EXECUTE FUNCTION check_seat_availability();


Триггер 3: Автоматический расчёт цены билета
sql
CREATE OR REPLACE FUNCTION calculate_ticket_price()
RETURNS TRIGGER AS $$
DECLARE
    session_start TIMESTAMP;
    session_base_price DECIMAL(10,2);
    user_birth DATE;
    user_age INTEGER;
    final_price DECIMAL(10,2);
BEGIN
    -- Получаем информацию о сеансе
    SELECT s.start_time, s.base_price INTO session_start, session_base_price
    FROM sessions s WHERE s.id = NEW.session_id;
    
    final_price := session_base_price;
    
    -- Скидка за утренний сеанс (до 12:00)
    IF EXTRACT(HOUR FROM session_start) < 12 THEN
        final_price := final_price * 0.8; -- 20% скидка
    END IF;
    
    -- Если пользователь указан, проверяем возрастные скидки
    IF NEW.user_id IS NOT NULL THEN
        SELECT birth_date INTO user_birth FROM users WHERE id = NEW.user_id;
        IF user_birth IS NOT NULL THEN
            user_age := EXTRACT(YEAR FROM AGE(session_start, user_birth));
            -- Студенческая скидка (18-25 лет)
            IF user_age BETWEEN 18 AND 25 THEN
                final_price := final_price * 0.9; -- дополнительно 10%
            END IF;
            -- Пенсионная скидка (>= 60 лет)
            IF user_age >= 60 THEN
                final_price := final_price * 0.7; -- 30% скидка
            END IF;
        END IF;
    END IF;
    
    NEW.price := final_price;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_calculate_ticket_price
    BEFORE INSERT ON tickets
    FOR EACH ROW
    EXECUTE FUNCTION calculate_ticket_price();


Триггер 4: Логирование изменений статуса билета
sql
CREATE OR REPLACE FUNCTION log_ticket_status_change()
RETURNS TRIGGER AS $$
BEGIN
    IF OLD.status IS DISTINCT FROM NEW.status THEN
        INSERT INTO ticket_status_log (ticket_id, old_status, new_status, changed_at, changed_by)
        VALUES (NEW.id, OLD.status, NEW.status, NOW(), 'system');
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_log_ticket_status_change
    AFTER UPDATE OF status ON tickets
    FOR EACH ROW
    EXECUTE FUNCTION log_ticket_status_change();


Триггер 5: Защита от удаления сеанса с проданными билетами
sql
CREATE OR REPLACE FUNCTION prevent_session_deletion_with_tickets()
RETURNS TRIGGER AS $$
BEGIN
    IF EXISTS (
        SELECT 1 FROM tickets
        WHERE session_id = OLD.id
          AND status IN ('booked', 'paid')
    ) THEN
        RAISE EXCEPTION 'Нельзя удалить сеанс с проданными или забронированными билетами';
    END IF;
    RETURN OLD;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_prevent_session_deletion
    BEFORE DELETE ON sessions
    FOR EACH ROW
    EXECUTE FUNCTION prevent_session_deletion_with_tickets();


Триггер 6: Дополнительная проверка при изменении статуса билета (например, нельзя отменить после начала сеанса)
sql
CREATE OR REPLACE FUNCTION check_ticket_cancellation_allowed()
RETURNS TRIGGER AS $$
DECLARE
    session_start TIMESTAMP;
BEGIN
    -- Если статус меняется на 'cancelled' или 'refunded'
    IF NEW.status IN ('cancelled', 'refunded') AND OLD.status IN ('booked', 'paid') THEN
        -- Получаем время начала сеанса
        SELECT start_time INTO session_start
        FROM sessions WHERE id = NEW.session_id;
        
        -- Если сеанс уже начался, запрещаем отмену (возврат)
        IF session_start <= NOW() THEN
            RAISE EXCEPTION 'Нельзя отменить билет после начала сеанса';
        END IF;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_check_ticket_cancellation
    BEFORE UPDATE OF status ON tickets
    FOR EACH ROW
    WHEN (NEW.status IN ('cancelled', 'refunded'))
    EXECUTE FUNCTION check_ticket_cancellation_allowed();
